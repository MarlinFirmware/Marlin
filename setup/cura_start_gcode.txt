;@@@@@ PREHEAT SECTION @@@@@
;
M140 S{material_bed_temperature_layer_0} ; set heat bed temperature without stopping
; extruder set later on (allowing bed to heat for 5 minutes)
;M104 S{material_print_temperature_layer_0} ; set extruder temperature without stopping
;
;@@@@@ BED LEVELING @@@@@
;
; https://www.3dmakerengineering.com/blogs/3d-printing/unified-bed-leveling-marlin
;
G91 ; set relative position mode for safe raise of extruder
G1 Z10 F500 ; raise extruder 10mm (relative) first !!
G90 ; reset to absolute position mode
G92 E0 ; Reset Extruder
M190 S{material_bed_temperature_layer_0} ; wait for heat bed to reach temperature
M104 120 ; pre-warm extruder while waiting for bed heating
M117 Waiting 5 minutes for bed to heat evenly
M118 E1 Waiting 5 minutes for bed to heat evenly
G4 S240 ; wait for 4 minutes for bed to heat evenly
M104 S{material_print_temperature_layer_0} ; set extruder temperature without stopping
G4 S60 ; wait for 1 more minute for bed to heat evenly
G28 ; pre-home while waiting for bed to heat up
G29 A ; Activate UBL in case it was disabled
G29 L1 ; Load the mesh stored in slot 1
M109 S{material_print_temperature_layer_0} ; wait for extruder to reach temperature
G29 J ; probe 3 points and "tilt" mesh to match probe results
;M420 S1; use bed leveling (done above)
;
;@@@@@ PRE-PRINT SECTION @@@@@
;
G1 Z2.0 F3000 ; Move Z Axis up little to prevent scratching of Heat Bed
G1 X0.1 Y20 Z0.3 F5000.0 ; Move to start position
G1 X0.1 Y200.0 Z0.3 F1500.0 E15 ; Draw the first line
G1 X0.4 Y200.0 Z0.3 F5000.0 ; Move to side a little
G1 X0.4 Y20 Z0.3 F1500.0 E30 ; Draw the second line
;
;G1 E-0.1 ; retract extruder 1mm (testing)
;
G92 E0 ; Reset Extruder
G1 Z2.0 F3000 ; Move Z Axis up little to prevent scratching of Heat Bed
G1 X5 Y20 Z0.3 F5000.0 ; Move over to prevent blob squish
